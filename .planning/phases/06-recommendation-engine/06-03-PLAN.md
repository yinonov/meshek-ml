       ```python
       """Safe persistence helpers for LightGBM recommendation model bundles.

       Security notes:
         - joblib is the scikit-learn recommended persistence format; it uses
           the pickle protocol under the hood. Bundles MUST only be written
           and read by the service itself (never uploaded by users). Plan 04's
           lifespan hook loads from MESHEK_MODEL_PATH which is a deploy-env
           variable, never caller-controlled.
         - load_model_bundle rejects paths that resolve outside the configured
           models root to defend against path traversal via a poisoned
           MESHEK_MODEL_PATH value.
       """
       from __future__ import annotations

       import os
       from pathlib import Path
       from typing import Any, TypedDict

       import joblib

       class ModelBundle(TypedDict):
           model: Any
           residual_std: float
           feature_cols: list[str]

       _REQUIRED_KEYS = ("model", "residual_std", "feature_cols")

       DEFAULT_MODELS_DIR = Path("models").resolve()

       def _models_root() -> Path:
           """Resolve the allowed models root.

           Priority: MESHEK_MODELS_DIR env var (if set) > DEFAULT_MODELS_DIR.
           Tests monkeypatch MESHEK_MODELS_DIR to tmp_path.
           """
           raw = os.environ.get("MESHEK_MODELS_DIR")
           if raw:
               return Path(raw).resolve()
           return DEFAULT_MODELS_DIR

       def _assert_within_root(path: Path) -> Path:
           resolved = path.resolve()
           root = _models_root()
           try:
               resolved.relative_to(root)
           except ValueError as exc:
               raise ValueError(
                   f"Model path {resolved} is outside the allowed models root {root}"
               ) from exc
           return resolved

       def save_model_bundle(bundle: ModelBundle, path: Path) -> Path:
           resolved = _assert_within_root(path)
           resolved.parent.mkdir(parents=True, exist_ok=True)
           joblib.dump(bundle, resolved)
           return resolved

       def load_model_bundle(path: Path) -> ModelBundle:
           resolved = _assert_within_root(path)
           if not resolved.exists():
               raise FileNotFoundError(f"Model bundle not found at {resolved}")
           obj = joblib.load(resolved)
           if not isinstance(obj, dict):
               raise ValueError(f"Model file {resolved} is not a ModelBundle dict")
           missing = [k for k in _REQUIRED_KEYS if k not in obj]
           if missing:
               raise ValueError(
                   f"Model bundle at {resolved} is missing required keys: {missing}"
               )
           return obj  # type: ignore[return-value]
       ```

       ```python
       """Offline LightGBM training entry point for Tier 3 inference."""
       from __future__ import annotations

       from pathlib import Path

       import numpy as np
       import pandas as pd

       from meshek_ml.forecasting.features import (
           add_calendar_features,
           add_lag_features,
           add_rolling_features,
       )
       from meshek_ml.forecasting.pipeline import FEATURE_COLS_TO_DROP, TARGET_COL
       from meshek_ml.forecasting.schema import (
           normalize_simulation_data,
           validate_demand_schema,
       )
       from meshek_ml.forecasting.tree_models import train_lightgbm
       from meshek_ml.recommendation.model_io import ModelBundle, save_model_bundle

       def train_and_save(output_path: Path, data: pd.DataFrame) -> ModelBundle:
           df = data.copy()
           if "realized_demand" in df.columns:
               df = normalize_simulation_data(df)
           df = validate_demand_schema(df)
           df["date"] = pd.to_datetime(df["date"])
           df = df.sort_values(["merchant_id", "product", "date"]).reset_index(drop=True)
           df = add_lag_features(df, target_col=TARGET_COL)
           df = add_rolling_features(df, target_col=TARGET_COL)
           df = add_calendar_features(df)
           df = df.dropna()
           if df.empty:
               raise ValueError(
                   "No training rows remain after feature engineering dropna; "
                   "need more history."
               )
           feature_cols = [c for c in df.columns if c not in FEATURE_COLS_TO_DROP]
           x_train = df[feature_cols]
           y_train = df[TARGET_COL].to_numpy()
           model = train_lightgbm(x_train, y_train)
           residual_std = float(np.std(y_train - model.predict(x_train)))
           bundle: ModelBundle = {
               "model": model,
               "residual_std": residual_std,
               "feature_cols": feature_cols,
           }
           save_model_bundle(bundle, output_path)
           return bundle
       ```

       ```python
       from typing import Any

       from meshek_ml.forecasting.features import (
           add_calendar_features,
           add_lag_features,
           add_rolling_features,
       )
       from meshek_ml.optimization.newsvendor import optimal_order_normal

       def tier_3_ml_forecast(
           merchant_id: str,
           sales: pd.DataFrame,
           model: Any,
           residual_std: float,
           feature_cols: list[str],
           underage_cost: float = 2.0,
           overage_cost: float = 1.0,
       ) -> RecommendationResponse:
           if sales.empty:
               raise ValueError("tier_3_ml_forecast requires non-empty sales history")
           df = sales.copy()
           df["date"] = pd.to_datetime(df["date"])
           df = df.sort_values(["merchant_id", "product", "date"]).reset_index(drop=True)
           df = add_lag_features(df, target_col="quantity")
           df = add_rolling_features(df, target_col="quantity")
           df = add_calendar_features(df)
           last_rows = df.sort_values("date").groupby("product", as_index=False).tail(1)
           # Pitfall 1: impute NaN features with per-column mean over last_rows
           x = last_rows.reindex(columns=feature_cols)
           x = x.fillna(x.mean(numeric_only=True)).fillna(0.0)
           mu = model.predict(x)
           recs: list[ProductRecommendation] = []
           for product, mean_demand in zip(last_rows["product"], mu):
               q = optimal_order_normal(
                   mean_demand=float(mean_demand),
                   std_demand=float(residual_std),
                   underage_cost=underage_cost,
                   overage_cost=overage_cost,
               )
               recs.append(ProductRecommendation(
                   product_id=str(product),
                   quantity=round(float(q), 2),
                   unit="kg",
               ))
           y_mean = float(sales["quantity"].mean()) or 1.0
           raw = 1.0 - (residual_std / y_mean)
           confidence = max(0.6, min(0.95, raw))
           return RecommendationResponse(
               merchant_id=merchant_id,
               recommendations=recs,
               reasoning_tier="ml_forecast",
               confidence_score=round(confidence, 6),
               generated_at=datetime.now(timezone.utc),
           )
       ```

| Boundary | Description |
|----------|-------------|
| MESHEK_MODEL_PATH env var → filesystem path | Deploy-operator-controlled; path traversal if unvalidated. |
| Model bundle file → deserialization | joblib uses the pickle protocol; malicious files would be RCE. |
| Sales DataFrame → LightGBM.predict | Data flows from Phase 5 MerchantStore which already schema-validates. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-6-08 | Tampering (path traversal) | load_model_bundle / save_model_bundle | mitigate | Both helpers call `_assert_within_root(path)` which uses `Path.resolve()` + `relative_to(root)` and raises ValueError on escape. The allowed root is the MESHEK_MODELS_DIR env var (defaults to `./models`), read fresh per call so tests and deployments can override it safely. Enforced by `test_rejects_path_outside_models_dir` and grep acceptance on `relative_to(root)`. |
| T-6-09 | Tampering (untrusted deserialization) | load_model_bundle | mitigate | Contract: the service only loads files it wrote itself to the deploy-controlled models root. Documented in the module docstring. Loader rejects dicts missing required keys so corrupted files fail loudly rather than silently returning junk. REC-05 (retrain endpoint) will revisit and harden — deferred to v1.2. |
| T-6-10 | Information Disclosure | residual_std leaks training-set variance | accept | Scalar aggregate; not PII. Documented in SUMMARY. |
| T-6-11 | DoS | Adversarial sales DataFrame with unbounded feature explosion | accept | Phase 5 MerchantStore schema-validates writes; tier_3 only reads from a validated store. v1.1 volumes are small. |
