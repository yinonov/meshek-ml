from __future__ import annotations
from enum import Enum

class Unit(str, Enum):
    KG = "kg"
    GRAM = "gram"
    UNIT = "unit"
    CRATE = "crate"

# Public: the canonical Hebrew unit-token → Unit map used by the parser.
UNIT_ALIASES: dict[str, Unit]  # all keys are already normalized (post normalize_text)

def normalize_text(text: str) -> str:
    """Apply the full D-07 pipeline:
       1. strip leading/trailing whitespace
       2. NFD-decompose and drop combining marks in U+0591..U+05C7 (Hebrew niqqud)
       3. fold final-letter forms: ך→כ, ם→מ, ן→נ, ף→פ, ץ→צ
       4. lowercase Latin characters
       5. collapse runs of internal whitespace to a single ASCII space
    Non-destructive otherwise; Hebrew base letters are preserved.
    """

def match_unit_token(token: str) -> Unit | None:
    """Look up `token` (already normalized) in UNIT_ALIASES. Return None on miss."""
