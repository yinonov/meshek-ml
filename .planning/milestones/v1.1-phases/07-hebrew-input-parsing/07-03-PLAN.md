```python
from __future__ import annotations
from dataclasses import dataclass
from typing import Literal

from meshek_ml.parsing.normalize import Unit
from meshek_ml.parsing.catalog import ProductCatalog

ParseErrorKind = Literal[
    "empty_input",
    "unknown_product",
    "missing_quantity",
    "bad_quantity",
    "ambiguous_quantity",
]

@dataclass(frozen=True)
class ParsedSale:
    product_id: str
    quantity: float
    unit: Unit
    raw_text: str

@dataclass(frozen=True)
class ParseError:
    kind: ParseErrorKind
    raw_text: str
    hint: str | None = None   # optional Hebrew-friendly message (D-15)

def parse_sales_line(text: str, catalog: ProductCatalog) -> ParsedSale | ParseError:
    """Parse a single Hebrew sales line. Returns a ParsedSale or a ParseError.
    Never raises for bad input (raises only on programmer error, e.g. None)."""

def parse_sales_lines(
    lines: list[str], catalog: ProductCatalog
) -> list[ParsedSale | ParseError]:
    """Convenience batch helper. Same semantics, per element."""
```

Algorithm (document in the module docstring):
  1. If `text` is empty/whitespace → `ParseError("empty_input")`.
  2. `norm = normalize_text(text)`.
  3. Extract all numeric tokens via regex `r"(?<!\S)(\d+(?:\.\d+)?)(?!\S)"` applied to `norm`.
     - 0 numbers → `ParseError("missing_quantity")`.
     - 2+ numbers → `ParseError("ambiguous_quantity")`.
     - exactly 1 number → parse as `float`; if `<= 0` → `ParseError("bad_quantity")`.
  4. Remove the numeric token from `norm` (preserving surrounding structure).
  5. Tokenize the remainder on whitespace. Walk a sliding window from `max(window_size)` = `catalog.max_alias_token_count` down to 1, across all start positions, and take the FIRST match that resolves via `catalog.resolve(" ".join(window))`. This enforces greedy longest-first (D-11).
  6. If no product matches → `ParseError("unknown_product")`.
  7. Tokens NOT consumed by the numeric or the product window become candidate unit tokens. For each leftover token, call `match_unit_token(token)`; first hit wins. If none match, use `catalog.get(product_id).default_unit`.
  8. Return `ParsedSale(product_id, quantity, unit, raw_text=text)`.
