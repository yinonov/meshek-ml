```python
# src/meshek_ml/parsing/catalog.py
from __future__ import annotations
from dataclasses import dataclass, field
from pathlib import Path
from types import MappingProxyType

from meshek_ml.parsing.normalize import Unit, normalize_text

class CatalogError(ValueError):
    """Raised when the YAML catalog is malformed or has duplicate aliases."""

@dataclass(frozen=True)
class Product:
    product_id: str                 # canonical lowercase ASCII slug (D-03)
    display_he: str                 # Hebrew display name
    display_en: str                 # English display name
    category: str
    default_unit: Unit
    aliases: tuple[str, ...]        # raw (pre-normalized) Hebrew aliases

@dataclass(frozen=True)
class ProductCatalog:
    products: tuple[Product, ...]
    # Mapping from normalized-alias -> product_id. Immutable.
    alias_index: MappingProxyType[str, str]

    def resolve(self, normalized_alias: str) -> str | None:
        """Return canonical product_id for a normalized alias, or None on miss."""

    def get(self, product_id: str) -> Product | None:
        """Return Product by canonical id, or None."""

    @property
    def max_alias_token_count(self) -> int:
        """Max whitespace-token count across all normalized aliases.
        Plan 03's greedy longest-first matcher caps its window at this value."""

def load_catalog(path: Path) -> ProductCatalog:
    """Load + validate YAML catalog. Raises CatalogError on duplicate aliases."""
```

YAML shape (per entry):
```yaml
products:
  - id: tomato
    display_he: "עגבנייה"
    display_en: "tomato"
    category: vegetable
    default_unit: kg
    aliases:
      - "עגבנייה"
      - "עגבניה"        # common misspelling (no yod doubling)
      - "עגבניות"       # plural
      - "עגבניות אדומות"  # common two-word variant
```
